Implementation of Deadlock Prevention Using Bankerâ€™s Algorithm

#include <stdio.h>
#include <stdbool.h>
int main() {
int n, m, i, j, k;
printf("Enter number of processes: ");
scanf("%d", &n);
printf("Enter number of resources: ");
scanf("%d", &m);
int alloc[n][m], max[n][m], avail[m];
printf("Enter allocation matrix:\n");
for (i = 0; i < n; i++)
for (j = 0; j < m; j++)
scanf("%d", &alloc[i][j]);
printf("Enter maximum need matrix:\n");
for (i = 0; i < n; i++)
for (j = 0; j < m; j++)
scanf("%d", &max[i][j]);
printf("Enter available resources:\n");
for (j = 0; j < m; j++)
scanf("%d", &avail[j]);
int f[n], ans[n], ind = 0;
for (k = 0; k < n; k++) f[k] = 0;
int need[n][m];
for (i = 0; i < n; i++)
for (j = 0; j < m; j++)
need[i][j] = max[i][j] - alloc[i][j];
// Safety algorithm
int y = 0;
for (k = 0; k < n; k++) {
for (i = 0; i < n; i++) {
if (f[i] == 0) {
bool flag = true;
for (j = 0; j < m; j++) {
if (need[i][j] > avail[j]) {
flag = false;
break;
}
}
if (flag) {
ans[ind++] = i;
for (y = 0; y < m; y++)
avail[y] += alloc[i][y];
f[i] = 1;
}
}
}
}
bool safe = true;
for (i = 0; i < n; i++) {
if (f[i] == 0) {
safe = false;
printf("System is NOT in a safe state.\n");
break;
}
}
if (safe) {
printf("System is in a SAFE state.\nSafe sequence: ");
for (i = 0; i < n - 1; i++)
printf("P%d -> ", ans[i]);
printf("P%d\n", ans[n - 1]);
}
return 0;
}
